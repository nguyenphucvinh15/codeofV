#include <stdio.h>

void nhap(int a[][50], int *m, int *n) {
		printf("Nhap so nguyen m, n > 0: ");
		scanf("%d%d", m, n);	
	int i, j;
	for(i = 1; i <= *m; i++) {
		for(j = 1; j <= *n; j++) {
			printf("a[%d][%d] = ", i, j);
			scanf("%d", &a[i][j]);
		}
	}
}

void trave(int a[][50], int m, int n) {
	int i, j;
	for(i = 1; i <= m; i++) {
		for(j = 1; j <= n; j++) {
			printf("%5d ", a[i][j]);
		}
		printf("\n");
	}
}


void chuyenVi(int a[][50], int b[][50], int m, int n) {
for(int i=1;i<=n;i++)
{
	for(int j=1;j<=m;j++)
	{
		b[i][j]=a[j][i];// bij = aji
	}
}
}
int main() {
	int a[50][50], b[50][50];
	int m, n;
	printf("Nhap ma tran A: ");
	nhap(a, &m, &n);
	chuyenVi(a, b, n, m);
	
	printf("======================\n");
	printf("Ma tran A: \n");
	trave(a, m, n);
	
	printf("======================\n");
	printf("Ma tran B: \n");
	trave(b, n, m);
	
	return 0;
}
